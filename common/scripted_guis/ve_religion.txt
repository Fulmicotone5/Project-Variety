ve_send_missionary = {
	scope = state
	saved_scopes = {
		country
	}
	is_shown = {
		NOT = {
			has_modifier = ve_missionary_mdf
		}
	}
	is_valid = {
		scope:country = {
			has_variable = avaible_missionaries
			avaible_missionaries >= 1
			NOT = {
				has_law = law_type:law_total_separation
			}
		}
		owner = scope:country
	}
	effect = {
		set_variable = selected_state_religion
		if = {
			limit = {
				has_variable = ethnocentrism_idea_5_var
			}
			add_modifier = {
				name = ve_missionary_mdf
				multiplier = 3
				months = -1
			}
		}
		else = {
			add_modifier = {
				name = ve_missionary_mdf
				months = -1
			}
		}
		scope:country = {
			change_variable = {
				name = avaible_missionaries
				subtract = 1
			}
		}
	}
}
