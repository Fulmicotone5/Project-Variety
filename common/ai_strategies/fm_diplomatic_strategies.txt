ai_strategy_unify_poland = {
	icon = "gfx/interface/icons/ai_strategy_icons/unification.dds"
	type = diplomatic
	will_form_power_bloc = {
		always = yes
	}
	undesirable_infamy_level = {
		value = 80
	}
	unacceptable_infamy_level = {
		value = 120
	}
	ideological_opinion_effect_mult = 0.5
	wargoal_maneuvers_fraction = {
		value = 0.2
	}
	diplomatic_play_neutrality = {
		value = 0
	}
	diplomatic_play_boldness = {
		value = 100
	}
	obligation_value = {
		value = 20
	}
	recklessness = {
		value = 2
	}
	aggression = {
		value = 0.5
		if = {
			limit = {
				scope:target_country = {
					can_form_nation = PUL
				}
			}
			add = 200
		}
		if = {
			limit = {
				scope:target_country = {
					any_scope_state = {
						OR = {
							state_region = s:STATE_GREATER_POLAND
							state_region = s:STATE_LESSER_POLAND
							state_region = s:STATE_EAST_GALICIA
							state_region = s:STATE_WEST_GALICIA
							state_region = s:STATE_MAZOVIA
						}
					}
				}
			}
			add = 200
		}
	}
	building_group_weights = {
		bg_army = 1.5
	}
	state_value = {
		if = {
			limit = {
				scope:target_state = {
					state_region = {
						is_homeland = cu:polish
					}
				}
				can_reach_target_state = yes
			}
			add = 500
		}
	}
	wargoal_scores = {
		annex_country = {
			if = {
				limit = {
					can_reach_target_country = yes
				}
				add = 300
			}
		}
	}
	secret_goal_scores = {
		befriend = {
			if = {
				limit = {
					NOT = {
						scope:target_country = {
							is_subject_of = root
						}
					} 
				}
				add = 50
				if = {
					limit = {
						can_reach_target_country = no
					}
					add = 150
				}
			}
		}
		conquer = {
			if = {
				limit = {
					NOT = {
						scope:target_country = {
							is_subject_of = root
						}
					}
					scope:target_country = {
						any_scope_state = {
							OR = {
								state_region = s:STATE_GREATER_POLAND
								state_region = s:STATE_LESSER_POLAND
								state_region = s:STATE_EAST_GALICIA
								state_region = s:STATE_WEST_GALICIA
								state_region = s:STATE_MAZOVIA
							}
						}
					}
					can_reach_target_country = yes
				}
				add = 500
			}
		}
	}
	possible = {
		is_subject = no
		country_has_primary_culture = cu:polish
	}
	weight = {
		value = 25
		add = {
			value = army_size
			multiply = 10
		}
		if = {
			limit = {
				NOT = {
					is_country_type = unrecognized
				}
			}
			multiply = 10
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_major_power
			}
			multiply = 5
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_regional_power
			}
			multiply = 3
		}
	}
}

ai_strategy_unify_ethiopia = {
	icon = "gfx/interface/icons/ai_strategy_icons/unification.dds"
	type = diplomatic
	will_form_power_bloc = {
		always = yes
	}
	undesirable_infamy_level = {
		value = 80
	}
	unacceptable_infamy_level = {
		value = 120
	}
	ideological_opinion_effect_mult = 0.5
	wargoal_maneuvers_fraction = {
		value = 0.2
	}
	diplomatic_play_neutrality = {
		value = 0
	}
	diplomatic_play_boldness = {
		value = 50
	}
	obligation_value = {
		value = 0
		if = {
			limit = {
				scope:target_country = {
					has_journal_entry = je_age_of_princes
				}
			}
			add = 25
		}
	}
	recklessness = {
		value = 1
	}
	aggression = {
		value = 0.5
		if = {
			limit = {
				scope:target_country = {
					can_form_nation = ETH
				}
			}
			add = 50
		}
		if = {
			limit = {
				scope:target_country = {
					any_scope_state = {
						OR = {
							state_region = s:STATE_AMHARA
							state_region = s:STATE_GONDER
						}
					}
				}
			}
			add = 20
		}
		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
			}
			multiply = 0.25
		}
	}
	building_group_weights = {
		bg_army = 1.5
	}
	state_value = {
		if = {
			limit = {
				scope:target_state = {
					state_region = {
						OR = {
							is_homeland = cu:amhara
							is_homeland = cu:oromo
							is_homeland = cu:tigray
						}
					}
				}
				can_reach_target_state = yes
			}
			add = 100
		}
	}
	wargoal_scores = {
		annex_country = {
			if = {
				limit = {
					scope:target_country = {
						has_journal_entry = je_age_of_princes
					}
					can_reach_target_country = yes
				}
				add = 300
			}
		}
	}
	secret_goal_scores = {
		befriend = {
			if = {
				limit = {
					NOT = {
						scope:target_country = {
							is_subject_of = root
						}
					}
					scope:target_country = {
						any_scope_state = {
							OR = {
								state_region = s:STATE_AMHARA
								state_region = s:STATE_GONDER
							}
						}
					}
				}
				add = 150
				if = {
					limit = {
						can_reach_target_country = no
					}
					add = 150
				}
			}
		}
		conquer = {
			if = {
				limit = {
					NOT = {
						scope:target_country = {
							is_subject_of = root
						}
					}
					scope:target_country = {
						any_scope_state = {
							OR = {
								state_region = s:STATE_AMHARA
								state_region = s:STATE_GONDER
							}
						}
					}
					can_reach_target_country = yes
				}
				add = 500
			}
		}
	}
	possible = {
		is_subject = no
		has_journal_entry = je_age_of_princes
	}
	weight = {
		value = 25
		add = {
			value = army_size
			multiply = 10
		}
		if = {
			limit = {
				NOT = {
					is_country_type = unrecognized
				}
			}
			multiply = 10
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_major_power
			}
			multiply = 5
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_regional_power
			}
			multiply = 3
		}
		# Delay the Ethiopian Thunderdome a bit
		if = {
			limit = {
				NOT = {
					has_technology_researched = centralization
				}
			}
			multiply = 0.1
		}
	}
}
