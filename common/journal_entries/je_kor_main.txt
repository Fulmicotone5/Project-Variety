je_kor_main = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_historical_content
	timeout = 3000

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	fail = {
		NOT = {
			has_law = law_type:law_monarchy
		}
	}

	on_complete = {
		trigger_event = {
			id = fm_kor.7
		}
		remove_modifier = diplomatic_plays_blocked
	}

	on_timeout = {
		trigger_event = {
			id = fm_kor.8
		}
		remove_modifier = diplomatic_plays_blocked
	}

	on_fail = {
		trigger_event = {
			id = fm_kor.8
		}
		remove_modifier = diplomatic_plays_blocked
	}

	current_value = {
		value = var:fm_kor_var
	}

	goal_add_value = {
		add = 5
	}

	progressbar = yes

	transferable = yes

	weight = 10000
	should_be_pinned_by_default = yes
}

je_kor_autoc = {

	group = je_group_historical_content

	complete = {
		OR = {
			has_law = law_type:law_oligarchy
			has_law = law_type:law_autocracy
		}
	}

	on_complete = {
		custom_tooltip = je_korea_advance
		change_variable = {
			name = fm_kor_var
			add = 1
		}
		hidden_effect = {
			trigger_event = {
				id = fm_kor.2
				days = 60

			}
		}
	}
	fail = {
		custom_tooltip = {
			text = fm_kor_mil
			has_variable = fm_kor_coup
		}
	}
	transferable = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
