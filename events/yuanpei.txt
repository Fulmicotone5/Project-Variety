fm_yua.7 = {
	type = country_event
	placement = ROOT
	title = fm_yua.7.t
	desc = fm_yua.7.d
	flavor = fm_yua.7.f
	duration = 3
	event_image = {
		video = "ep1_clandestine_meeting"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	immediate = {
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_armed_forces
				exist = leader
			}
			leader = {
				save_scope_as = armed_forces_leader
			}
		}
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_landowners
				exist = leader
			}
			leader = {
				save_scope_as = landowners_leader
			}
		}
		random_interest_group = {
			limit
			{
				is_interest_group_type = ig_petty_bourgeoisie
				exist = leader
			}
			leader = {
				save_scope_as = petty_leader
			}
		}
	}
	option = {
		name = fm_yua.7.a
		add_loyalists = {
			value = 0.5
			pop_type = academics
		}
		add_loyalists = {
			value = 0.4
			pop_type = bureaucrats
		}
		hidden_effect = {
			scope:landowners_leader = {
				set_ideology = ideology:ideology_positivist
			}
			scope:petty_leader = {
				set_ideology = ideology:ideology_positivist
			}
			scope:armed_forces_leader = {
				set_ideology = ideology:ideology_positivist
			}
		}
		set_variable = {
			name = yua_yua_authpot
			value = 0
		}
	}
}
